完整流程（伪代码）
templen = len
avg = limit
for j in 0..N-1:
    templen2 = 0
    for i in 0..templen-1:
        if in[i] >= avg:
            midIn[templen2++] = in[i]
    templen = templen2
    if templen < 10:
        return 0                  // 无信号
    for k in 0..templen-1:
        in[k] = midIn[k]         // 收缩 in 数组
        sum += midIn[k]
    avg = sum / templen           // 抬升阈值到当前均值
return select_sort(in, templen)    // 截尾均值（去 2 大 + 去 2 小）
